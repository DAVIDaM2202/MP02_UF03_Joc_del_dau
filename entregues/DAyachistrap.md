CREATE FUNCTION dbo.Melsquedo (@nJugador AS INT, 
                               @punts AS INT) 
returns BIT 
AS 
  BEGIN 
      DECLARE @siono BIT; 

      IF @punts BETWEEN 0 AND 3 
        BEGIN 
            SET @siono=0 
        END; 
      ELSE 
        BEGIN 
            SET @siono=1 
        END; 

      RETURN @siono 
  END; 

go
